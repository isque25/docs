---
title: "Extract API"
api: "GET /api/v1/extract"
description: "Extract content from webpages with advanced parsing"
---

The extract endpoint retrieves and processes content from web pages, providing clean text, images, and links in a structured format.

## Query Parameters

<ParamField query="url" type="string" required>
  URL to extract content from
</ParamField>

<ParamField query="include_images" type="boolean" default="false">
  Include images in the response
</ParamField>

<ParamField query="include_links" type="boolean" default="false">
  Include internal links found on the page in the response
</ParamField>

<ParamField query="extract_depth" type="string" default="basic">
  Depth of extraction. 'advanced' retrieves more data including tables and
  embedded content.
</ParamField>

## Response

<ResponseField name="results" type="ExtractResult[]">
  List of successful extractions
  <Expandable title="ExtractResult properties">
    <ResponseField name="url" type="string">
      The URL from which content was extracted
    </ResponseField>

    <ResponseField name="raw_content" type="string">
      The full content extracted from the page
    </ResponseField>

    <ResponseField name="images" type="string[]">
      List of image URLs extracted from the page
    </ResponseField>

    <ResponseField name="links" type="string[]">
      List of internal links found on the page
    </ResponseField>

    <ResponseField name="cache_hit" type="boolean">
      Indicates if this result was served from cache
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="failed_results" type="FailedResult[]">
  List of failed extractions
  <Expandable title="FailedResult properties">
    <ResponseField name="url" type="string">
      The URL that failed to be processed
    </ResponseField>

    <ResponseField name="error" type="string">
      Error message describing why the URL couldn't be processed
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="response_time" type="number">
  Time in seconds it took to complete the request
</ResponseField>

## Bulk Extraction (POST Method)

<Note>
  For extracting content from multiple URLs at once, use the POST version of
  this endpoint.
</Note>

```typescript
const response = await client.post({
  url: "/api/v1/extract",
  body: {
    urls: ["https://example.com", "https://another-site.com"],
    include_images: true,
    include_links: true,
    extract_depth: "advanced",
  },
});
```

## TypeScript Example

```typescript
import { client } from "@agent-toolkit/client-ts";
import type { ExtractResponse } from "@agent-toolkit/client-ts";

// Configure the client
client.setConfig({
  baseUrl: "https://api.agenttoolkit.ai",
  headers: {
    "Content-Type": "application/json",
    "X-API-Key": "your-api-key",
  },
});

// Extract content from a website
const response = await client.get<ExtractResponse>({
  url: "/api/v1/extract",
  query: {
    url: "https://www.example.com",
    include_images: true,
    include_links: true,
    extract_depth: "advanced",
  },
});

// Display extracted content
if (response.data?.results?.[0]) {
  const result = response.data.results[0];
  console.log("Extracted Content:", result.raw_content);
  console.log("Images:", result.images?.length || 0);
  console.log("Links:", result.links?.length || 0);
}
```
